{{- /* github.html — repo card with live build-time metadata. Fetches the GitHub API once per build (cached; ~60 req/hr unauthenticated is plenty for a handful of repos). Degrades gracefully: if the fetch fails the card still renders with the repo name and a working link, just no stats. `showThumbnail` is accepted for content compatibility but unused — the slate card is text-only by design. Usage: {{< github repo="joseph-wardle/javelin" >}} */ -}} {{- $repo := or (.Get "repo") (.Get 0) -}} {{- with $repo -}} {{- $colors := dict "Rust" "#dea584" "C++" "#f34b7d" "C" "#555555" "Python" "#3572A5" "Go" "#00ADD8" "TypeScript" "#3178c6" "JavaScript" "#f1e05a" "Shell" "#89e051" "GLSL" "#5686a5" "HLSL" "#aace60" "Slang" "#fe8019" "CMake" "#DA3434" "C#" "#178600" "Lua" "#000080" "HTML" "#e34c26" -}} {{- $name := . -}}{{- $url := printf "https://github.com/%s" . -}} {{- $desc := "" -}}{{- $lang := "" -}}{{- $stars := 0 -}}{{- $forks := 0 -}}{{- $ok := false -}} {{- $api := printf "https://api.github.com/repos/%s" . -}} {{- $r := try (resources.GetRemote $api (dict "headers" (dict "Accept" "application/vnd.github+json"))) -}} {{- with $r.Err -}} {{- warnf "github shortcode %q: %s" $repo . -}} {{- else -}} {{- with $r.Value -}} {{- $d := unmarshal .Content -}} {{- $name = $d.full_name -}}{{- $url = $d.html_url -}} {{- $desc = $d.description -}}{{- $lang = string $d.language -}} {{- $stars = $d.stargazers_count -}}{{- $forks = $d.forks_count -}}{{- $ok = true -}} {{- end -}} {{- end -}} {{- $lc := index $colors $lang | default "#7c6f64" -}} {{ $name }} {{- with $desc }}{{ . }}{{ end }} {{- if $ok }} {{- with $lang }}{{ . }}{{ end }} ★ {{ lang.FormatNumberCustom 0 $stars }} ⮀ {{ lang.FormatNumberCustom 0 $forks }} {{- else }} View on GitHub → {{- end }} {{- end -}}
